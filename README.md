# Ip-track
Una simple herramienta para rastrear Ip programada en Python
# Instalacion

apt install git python python2

cd $HOME

git clone https://github.com/Euronymou5/Ip-track

cd Ip-track

chmod +x requirements.sh Tracker.py

bash requirements.sh

python2 Tracker.py
